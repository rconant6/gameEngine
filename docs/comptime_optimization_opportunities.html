pub const shape_types = blk: {
    const decls = @typeInfo(Shapes).@"struct".decls;
    const decls_len = decls.len;
    const point_types_count = point_types.len;  // 2: WorldPoint, ScreenPoint
    const total_size = decls_len * point_types_count;
    var types: [total_size]type = undefined;
    for (decls, 0..) |decl, i| {
        if (@typeInfo(@TypeOf(@field(Shapes, decl.name))) == .@"fn") {
            for (point_types, 0..) |point_type, j| {
                types[i * point_types_count + j] = @field(Shapes, decl.name)(point_type);
            }
        }
    }
    break :blk types;
};
Status: Already using comptime effectively. The code correctly specializes generic shape functions for different coordinate spaces at compile time.

3.3 Input Type Validation (Enhancement)

Medium Impact Low Effort
src/platform/Input.zig (lines 24-54)

Current Pattern:

const SupportedInputTypes = .{ KeyCode, MouseButton };

pub fn validateSupportedType(comptime T: type) void {
    inline for (SupportedInputTypes) |SupportedType| {
        if (T == SupportedType) return;
    }
    @compileError("Input type " ++ @typeName(T) ++ " not handled in Input");
}

pub inline fn isDown(self: *const Self, input: anytype) bool {
    const T = @TypeOf(input);
    validateSupportedType(T);
    if (T == KeyCode) return self.keyboard.isDown(input);
    if (T == MouseButton) return self.mouse.buttons.isDown(input);
    unreachable;
}
Status: Good runtime type checking with comptime validation.

Enhancement Opportunity:

The runtime type switching could be comptime-specialized:

pub inline fn isDown(self: *const Self, comptime T: type, input: T) bool {
    validateSupportedType(T);
    if (T == KeyCode) return self.keyboard.isDown(input);
    if (T == MouseButton) return self.mouse.buttons.isDown(input);
    unreachable;
}

This allows the Zig compiler to eliminate dead code paths per call site.
