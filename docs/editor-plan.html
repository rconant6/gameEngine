    scene_file:       ?*SceneFile,   // parsed AST — null until file opened

🖼 main.zig structure

pub fn main(init: std.process.Init) !void {
    var app = try App.init(gpa, io, .{
        .title = "Scene Editor",
        .width = 1920, .height = 1080,
    });
    defer app.deinit();

    var state = EditorState.init(gpa);
    defer state.deinit();

    var ui_layer = UILayer.init(gpa);
    ui_layer.addView("hierarchy", .{ .x=0,    .y=0, .width=280,  .height=1080 }, Hierarchy.build, .{});
    ui_layer.addView("inspector", .{ .x=1620, .y=0, .width=300,  .height=1080 }, Inspector.build, .{});
    // Viewport is drawn directly each frame (not a UI view)

    while (app.isRunning()) {
        try app.beginFrame();
        app.renderer.clear();

        handleInput(&state, &app);   // Cmd+O, Cmd+S, scroll zoom
        drawViewport(&state, &app);  // scene preview
        ui_layer.update(&state, mouse_x, mouse_y, left_down, left_up);
        ui_layer.render(&app.renderer, &font, ctx);

        try app.endFrame();
    }
}

Declaration headers

[TemplateName:template]

  [ChildEntity:entity]
Value formats
vec2:   {1.0, 2.0}
vec3:   {1.0, 2.0, 0.0}
color:  #RRGGBB
bool:   true / false
f32:    1.0  (always decimal)
string: "value"
Special: Sprite/Collider (has shape type)
[Sprite:circle]         // not [Sprite]
  radius:f32 2.0
  fill_color:color #00FF00

🔄 Mutation — wiring slider → AST

// After ui_layer.update() each frame in main.zig:
fn flushInspectorChanges(ui: *UILayer, state: *EditorState) void {
    const entity = state.getSelectedEntity() orelse return;
    for (entity.components) |comp| {
        const props = getProperties(&comp);
        for (props) |*prop| {
            // slider widget stores its value under the key "entity.comp.prop"
            const id = makeId(entity.name, comp.name(), prop.name);
            if (ui.getState("inspector", id)) |ws| {
                if (ws.value != astValue(prop)) {
                    prop.value = valueFromWidgetState(ws, prop.type_annotation);
                    state.dirty = true;
                }
            }
        }
    }
}

💾 EditorState.loadFile / save

pub fn loadFile(self: *Self, path: []const u8) !void {
    const src = try fs.cwd()
        .readFileAlloc(self.gpa, path, 1024*1024);
    defer self.gpa.free(src);

    if (self.scene_file) |sf| {
        sf.deinit(self.gpa);
        self.gpa.destroy(sf);
    }
    self.scene_file = try parser.parse(self.gpa, src, path);
    self.scene_path  = try self.gpa.dupe(u8, path);
    self.selected    = null;
    self.dirty       = false;
}

pub fn save(self: *Self) !void {
    const file = try fs.cwd().createFile(self.scene_path, .{});
    defer file.close();
    try writer.serialize(self.scene_file.?, file.writer());
    self.dirty = false;
}
